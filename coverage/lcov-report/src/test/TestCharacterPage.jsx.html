import React, { useState } from 'react';
import { CharacterTemplate } from '../components/CharacterTemplate';
import { SkillTree } from '../components/SkillTree';
import { SkillModal } from '../components/SkillModal';
 
// Test character data
const testCharacter = {
  meta: {
    character_id: "test-character-1",
    level: 1,
    available_skill_points: 10
  },
  primary_scores: {
    Strength: 1,
    Dexterity: 1,
    Constitution: 1,
    Intelligence: 1,
    Wisdom: 1,
    Charisma: 1,
  },
  secondary_scores: {
    Willpower: 0,
    Spirit: 0,
    Arcane: 0,
    Presence: 0,
  },
  skills: {
    Strength: {
      "Anchor Stance": {
        description: "Plant yourself solidly to resist being pushed or knocked down.",
        boost: ["armor:heavy"],
        effective_value: 0,
      }
    },
    Arcane: {
      "Tier 1": {
        "Frost": {
          "Frost Shard": 0,
          "Chill Skin": 0
        }
      }
    }
  }
};
 
export const TestCharacterPage = () => {
  const [character, setCharacter] = useState(testCharacter);
  const [selectedStat, setSelectedStat] = useState(null);
  const [selectedSkill, setSelectedSkill] = useState(null);
 
  const handleStatSelect = (stat) => {
    setSelectedStat(stat);
    setSelectedSkill(null);
  };
 
  const handleSkillClick = (skill) => {
    setSelectedSkill(skill);
  };
 
  const handleCloseModal = () => {
    setSelectedSkill(null);
  };
 
  const handleUpgradeSkill = (stat, tier, category, skillName, amount) => {
    const available = character.meta.available_skill_points;
    const toAdd = Math.min(amount, available);
 
    if (toAdd <= 0) return;
 
    setCharacter(prev => {
      const updated = {...prev};
      
      // Handle both primary and mystical skills
      if (tier && category) {
        // Mystical skill
        if (!updated.skills[stat]) updated.skills[stat] = {};
        if (!updated.skills[stat][tier]) updated.skills[stat][tier] = {};
        if (!updated.skills[stat][tier][category]) updated.skills[stat][tier][category] = {};
        
        updated.skills[stat][tier][category][skillName] = 
          (updated.skills[stat][tier][category][skillName] || 0) + toAdd;
      } else {
        // Primary skill
        if (!updated.skills[stat]) updated.skills[stat] = {};
        updated.skills[stat][skillName] = (updated.skills[stat][skillName] || 0) + toAdd;
      }
 
      updated.meta.available_skill_points -= toAdd;
      return updated;
    });
  };
 
  const handleDowngradeSkill = (stat, tier, category, skillName, amount) => {
    setCharacter(prev => {
      const updated = {...prev};
      let currentValue;
      
      // Handle both primary and mystical skills
      if (tier && category) {
        // Mystical skill
        currentValue = updated.skills[stat]?.[tier]?.[category]?.[skillName] || 0;
        const toRemove = amount === 'reset' ? currentValue : Math.min(amount, currentValue);
        
        if (toRemove <= 0) return prev;
        
        updated.skills[stat][tier][category][skillName] = currentValue - toRemove;
        updated.meta.available_skill_points += toRemove;
      } else {
        // Primary skill
        currentValue = updated.skills[stat]?.[skillName] || 0;
        const toRemove = amount === 'reset' ? currentValue : Math.min(amount, currentValue);
        
        if (toRemove <= 0) return prev;
        
        updated.skills[stat][skillName] = currentValue - toRemove;
        updated.meta.available_skill_points += toRemove;
      }
 
      return updated;
    });
  };
 
  return (
    <div className="test-page">
      <div className="controls">
        <h1>Test Character Page</h1>
        <button onClick={() => setCharacter(testCharacter)}>Reset Character</button>
        <div>Available Points: {character.meta.available_skill_points}</div>
      </div>
 
      <div className="character-page">
        <div className="character-info">
          <CharacterTemplate characterId={character.meta.character_id} />
        </div>
 
        <div className="skill-section">
          <div className="stat-buttons">
            {Object.keys({...character.primary_scores, ...character.secondary_scores}).map(stat => (
              <button
                key={stat}
                onClick={() => handleStatSelect(stat)}
                className={selectedStat === stat ? 'active' : ''}
              >
                {stat}
              </button>
            ))}
          </div>
 
          {selectedStat && (
            <div className="skill-tree-container">
              <SkillTree
                characterData={character}
                statType={selectedStat}
                onSkillClick={handleSkillClick}
              />
            </div>
          )}
 
          {selectedSkill && (
            <SkillModal
              skill={selectedSkill}
              availablePoints={character.meta.available_skill_points}
              onClose={handleCloseModal}
              onUpgrade={handleUpgradeSkill}
              onDowngrade={handleDowngradeSkill}
            />
          )}
        </div>
      </div>
 
      <style jsx>{`
        .test-page {
          padding: 20px;
          background: #1a1a1a;
          color: white;
          min-height: 100vh;
        }
 
        .controls {
          margin-bottom: 20px;
          padding: 20px;
          background: #222;
          border-radius: 8px;
        }
 
        .character-page {
          display: grid;
          grid-template-columns: 300px 1fr;
          gap: 20px;
        }
 
        .character-info {
          background: #222;
          border-radius: 8px;
          padding: 16px;
        }
 
        .skill-section {
          display: flex;
          flex-direction: column;
          gap: 20px;
        }
 
        .stat-buttons {
          display: flex;
          flex-wrap: wrap;
          gap: 8px;
        }
 
        .stat-buttons button {
          padding: 8px 16px;
          border-radius: 6px;
          border: none;
          background: #333;
          color: white;
          cursor: pointer;
          transition: background 0.2s;
        }
 
        .stat-buttons button:hover {
          background: #444;
        }
 
        .stat-buttons button.active {
          background: #0af;
        }
 
        .skill-tree-container {
          flex: 1;
          min-height: 600px;
          background: #222;
          border-radius: 8px;
          overflow: hidden;
        }
      `}</style>
    </div>
  );
};
